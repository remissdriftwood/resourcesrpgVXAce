#===============================================================================
# ** Blunt skill processor
#===============================================================================
def blunt_processor(user, targets)
    
  #---------------------------------------------------------------------------
  # Blunt MP cost
  #---------------------------------------------------------------------------
  $game_actors[user].mp -= $game_actors[user].mmp
  
  #---------------------------------------------------------------------------
  # Hefty spin
  #---------------------------------------------------------------------------
  if battle_info_used.id == 6 then 
    $game_actors[user].add_state(28) else end
    
    
end
