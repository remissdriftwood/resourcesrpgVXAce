class Window_BattleStatus < Window_Selectable
  # Overrides the HP drawing for the battle status window
  def draw_actor_hp(actor, x, y, width = 124)
    draw_gauge(x, y, width, actor.hp_rate, hp_gauge_color1, hp_gauge_color2)
    
    change_color(normal_color)
    hp_string = "#{actor.hp}/#{actor.mhp}"
    
    # Draw the numbers across the full width without the "HP" label
    draw_text(x, y, width, line_height, hp_string, 2)
  end

  # Overrides the MP drawing for the battle status window
  def draw_actor_mp(actor, x, y, width = 124)
    draw_gauge(x, y, width, actor.mp_rate, mp_gauge_color1, mp_gauge_color2)
    
    change_color(normal_color)
    mp_string = "#{actor.mp}/#{actor.mmp}"
    
    # Draw the numbers across the full width without the "MP" label
    draw_text(x, y, width, line_height, mp_string, 2)
  end
end
